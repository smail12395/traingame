🚆 لعبة سكك القطارات

from js import document import random import asyncio GRID_SIZE = 10 trains = [] exits = {} COLORS = ['red', 'blue', 'green'] # توليد الشبكة async def generate_grid(): grid = document.getElementById("game-grid") grid.innerHTML = "" for row in range(GRID_SIZE): for col in range(GRID_SIZE): div = document.createElement("div") div.classList.add("cell") if random.random() < 0.4: div.classList.add("track") if random.random() < 0.2: div.classList.add("switch") div.innerHTML = "↕" def toggle(event, el=div): el.innerHTML = "↔" if el.innerHTML == "↕" else "↕" div.addEventListener("click", toggle) grid.appendChild(div) # ضع المخارج for i, color in enumerate(COLORS): exit_cell = grid.children[(GRID_SIZE-1)*GRID_SIZE + i*3] exit_cell.classList.add("exit") exit_cell.classList.add(color) exits[color] = exit_cell # أضف القطارات await add_trains(grid) # إنشاء القطارات async def add_trains(grid): for i, color in enumerate(COLORS): cell = grid.children[i] train = document.createElement("div") train.classList.add("train") train.classList.add(color) cell.appendChild(train) trains.append({"element": train, "color": color, "position": i}) await move_trains(grid) # تحريك القطارات async def move_trains(grid): while True: await asyncio.sleep(1) for train in trains: current = train["position"] next_pos = current + GRID_SIZE if next_pos >= GRID_SIZE * GRID_SIZE: continue next_cell = grid.children[next_pos] if "exit" in next_cell.className: if train["color"] in next_cell.className: next_cell.appendChild(train["element"]) train["position"] = next_pos print(f"✅ القطار {train['color']} وصل لمخرجه الصحيح") else: print(f"❌ القطار {train['color']} دخل مخرج خاطئ!") elif "track" in next_cell.className: next_cell.appendChild(train["element"]) train["position"] = next_pos await generate_grid()
